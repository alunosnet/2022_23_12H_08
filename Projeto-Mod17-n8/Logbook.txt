create table utilizadores(
	id int identity primary key,
	nome varchar(100) not null,
	nif varchar(9) not null,
	sal int,
	password varchar(64) not null,
	email varchar(100) not null unique check (email like '%@%.%'),
	perfil int not null check (perfil in ('0','1')),
	lnkRecuperar varchar(36),
	data_lnk date,
	estado int not null

)

CREATE TABLE Roupas
(
	[nroupa] INT NOT NULL PRIMARY KEY identity,
	tipo_roupa varchar(100) not null,
	marca varchar(100) not null,
	modelo varchar(100) not null,
	genero varchar (1) not null check(genero in ('m','f')),
	preco decimal(5,2) not null,
	quantidade int not null check(quantidade >= 0)
)

create table vendas(
	nvenda int identity primary key,
	nroupa int references roupas(nroupa),
	idutilizador int references utilizadores(id),
	data_venda date default(getdate()),
	v_preco decimal(5,2) not null,

)
